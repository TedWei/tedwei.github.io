buildWithGulp

2015-08-05

gulp是一个前端构建工具。使用它你可以将以前自己手工干的活儿进行自动化构建。
例如自动刷新页面，检查JS语法，压缩等，是个好工具。最近准备在组里推广下，在此写个教程出来。（如有遗失或错误，请黑）
gulp是运行于 Node 下的组件
所以需要先安装Node

之后请安装 gulp

$ npm install -g gulp //Install gulp globally
$ npm install --save-dev gulp //Install gulp in your project devDependencies

在项目文件夹根目录下创建gulpfile.js

var gulp = require('gulp');
gulp.task('default',function(){
    console.log("gulp running");
})

运行gulp

$ gulp

gulp的API文档
支持如下gulp.src | gulp.dest | gulp.task | gulp.watch

下面是一个基本的支持liveload,CSS压缩，JS压缩，CSS自动添加浏览器前缀支持.
需要引入以下gulp插件

$ npm install --save-dev gulp-autoprefixer gulp-minify-css gulp-jshint gulp-uglify gulp-rename gulp-cached gulp-connect

gulpfile.js 完整文件如下

'use strict';

// 载入外挂
var gulp = require('gulp'),
  autoprefixer = require('gulp-autoprefixer'),
  minifycss = require('gulp-minify-css'),
  jshint = require('gulp-jshint'),
  uglify = require('gulp-uglify'),
  rename = require('gulp-rename'),
  cache = require('gulp-cached'),
  connect = require('gulp-connect');


// 样式
gulp.task('compressCSS', function() {
  return gulp.src(['module/**/*.css', '!module/**/*.min.css'], {
      base: "./"
    })
    .pipe(autoprefixer({
      browser: ['last 2 version', 'safari 5', 'ie 8', 'ie 9', 'opera 12.1', 'ios 6', 'android 4']
    }))
    .pipe(cache('compressCSSing'))
    .pipe(minifycss())
    .pipe(rename({
      suffix: '.min'
    }))
    .pipe(gulp.dest('.'))
    .pipe(connect.reload());
});


//压缩脚本
gulp.task('compressjs', function() {
  return gulp.src(['module/**/*.js', '!module/**/*.min.js'], {
      base: "./"
    })
    .pipe(cache('compressJSing'))
    .pipe(uglify())
    .pipe(rename({
      extname: '.min.js'
    }))
    .pipe(gulp.dest('.'))
    .pipe(connect.reload());
})


gulp.task('html', function() {
  return gulp.src('module/**/*.html')
    .pipe(connect.reload());
})


// 预设任务
gulp.task('default', function() {
  gulp.start('server');
});

gulp.task('dev', ['server', 'watch']);

gulp.task('server', function() {
  connect.server({
    livereload: true,
    port: 3000
  })
})

// 看守
gulp.task('watch', function() {

  gulp.watch(['module/**/*.js', '!module/**/*.min.js'], ['compressjs']);

  gulp.watch(['module/**/*.css', '!module/**/*.min.css'], ['compressCSS']);

  gulp.watch(['module/**/*.html'], ['html']);

});
