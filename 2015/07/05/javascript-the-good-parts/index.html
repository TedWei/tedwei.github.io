读书笔记—javascript the good parts

为 Function 添加 method 

Function.prototype.method = function(name,func){
    if (!this.prototype[name]){
        this.prototype[name] = func;
    }
}

//can use this to delete method 
delete Function.prototype[name]

使用method 为原型添加以下方法

Number.method('integer',function(){
    return Math[this < 0 ? 'ceil' : 'floor'](this);
});

console.log((-10 /3 ).integer());

String.method('trim',function(){
    return this.replace(/^\s+|\s+$/g,'');
})
//or use thie RegExp (/\s+/g,' ') to replace all space ,only keep one
console.log("   zhang   san de hao   ".trim());

递归的经典例子：汉诺塔

var hanoi = function(disc,src,aux,dst){
    if (disc > 0){
        hanoi(disc-1,src,dst,aux);
        console.log('Move disc '+disc+' from '+src+' to '+dst);
        hanoi(disc-1,aux,src,dst);
    }
}
hanoi(3,'Src','Aux','Dst')

遍历子节点

var walk_the_childrens = function walk(node){
    if (!nodes){
        var nodes = [];
    }
    var walkIter = function(node){
        if (node == null){
            return 
        }
        node = node.firstChild;

        while(node){
            nodes.push(node);
            walkIter(node);
            node = node.nextSibling;
        }
        return nodes;
    }

    return walkIter(node);

}
