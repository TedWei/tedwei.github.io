ackerman in scheme and javascript
scheme

//scheme edition
(define (A x y)
  (cond ((= y 0) 0)
        ((= x 0) (* 2 y))
        ((= y 1) 2)
        (else (A (- x 1)
                 (A x (- y 1))))))

javascript

//javascript edition
var ackermann = function(x,y){
    if (y === 0){
        return 0;
    }else if (x === 0){
        return 2*y;
    }else if (y === 1){
        return 2;
    }else{
        return ackermann(x-1,ackermann(x,y-1));
    }
}

Fibonacci in scheme and javascript
scheme

//树形递归
(define (fib n)
    (cond ((= n 0) 0)
          ((= n 1) 1)
          (else (+ (fib (- n 1))
                   (fib (- n 2))))))

//迭代
(define (fib n)
    (fib-iter 1 0 n))
(define (fib-iter a b counter)
    (if (= counter 0)
        b
        (fib-iter (+ a b) a (- counter 1))))

javascript

//树形递归
var fib = function(n){
    if (n === 0){
        return 0;
    }else if (n === 1){
        return 1;
    }else{
        return fib(n-1)+fib(n-2);
    }
}

//迭代
// use Camel-Case because javascript doesn't support use - for variable's name
var fib = function(n){
    var fibIter = function(a,b,counter){
        if (counter === 0){
            return b;
        }else{
            return fibIter(a+b,a,counter-1)
        }
    }
    return fibIter(1,0,n);
}
